64 def pytest_configure(config):
65 htmlpath = config.getoption("htmlpath")
66 if htmlpath:
67 missing_css_files = []
68 for csspath in config.getoption("css"):
69 if not Path(csspath).exists():
70 missing_css_files.append(csspath)
71  
72 if missing_css_files:
73 oserror = (
74 f"Missing CSS file{'s' if len(missing_css_files) > 1 else ''}:"
75 f" {', '.join(missing_css_files)}"
76 )
77 raise OSError(oserror)
78  
79 if not hasattr(config, "workerinput"):
80 # prevent opening htmlpath on worker nodes (xdist)
81 config._html = HTMLReport(htmlpath, config)
82 config.pluginmanager.register(config._html)
83  
84  
85 def pytest_unconfigure(config):
86 html = getattr(config, "_html", None)
87 if html:
88 del config._html
89 config.pluginmanager.unregister(html)
